# models/account_move.py
@api.constrains('ncf')
def _check_ncf_format(self):
    """Valida formato NCF según normativa DGII"""
    ncf_regex = r'^[A-Z]{1}\d{2}(0[1-9]|1[0-2])(0[1-9]|[12][0-9]|3[01])\d{8}$'
    if not re.match(ncf_regex, self.ncf):
        raise ValidationError(_"Formato NCF inválido"))
